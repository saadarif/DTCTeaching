    var_$x_bar <- c(var_$x_bar, mean(smp))
    hist(var_$x_bar, probability = TRUE, col = "red", xlim = c(-4, 
        4), xlab = "x / x_bar", main = "", ylim = c(0, 2.2))
    points(mean(smp), 0, pch = 19, cex = 1.5, col = "black")
